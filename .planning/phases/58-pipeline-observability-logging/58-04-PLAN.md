```typescript
export function collectSeamsFromJsonl(tmpRoot: string, campaignId: string): Set<string> {
  const logsDir = join(tmpRoot, "campaigns", campaignId, "logs");
  const files = readdirSync(logsDir).filter((f) => /^turn-\d+-[0-9a-f]{8}\.jsonl$/.test(f));
  const seams = new Set<string>();
  for (const f of files) {
    for (const line of readFileSync(join(logsDir, f), "utf-8").trim().split("\n")) {
      const rec = JSON.parse(line);
      if (rec.event) seams.add(rec.event);
    }
  }
  return seams;
}
```

    ```typescript
      "sse.emit", "prompt.assembled", "storyteller.hidden.stream",

    export function collectSeamsFromJsonl(tmpRoot: string, campaignId: string): Set<string> {
      const logsDir = join(tmpRoot, "campaigns", campaignId, "logs");
      const files = readdirSync(logsDir).filter((f) => /^turn-\d+-[0-9a-f]{8}\.jsonl$/.test(f));
      const seams = new Set<string>();
      for (const f of files) {

          try {
            const rec = JSON.parse(line);

          } catch { /* skip malformed */ }
    ```

    ```typescript
    import { describe, it, expect, beforeEach, afterEach, vi } from "vitest";
    ```

    ```typescript
    function normalizeSse(raw: Buffer): Buffer {
      let s = raw.toString("utf8");
      // Normalize timestamps and UUIDs so byte-compare is stable
      s = s.replace(/"timestamp":\s*\d+/g, '"timestamp":0');
      s = s.replace(/[0-9a-f]{8}-[0-9a-f]{4}-[0-9a-f]{4}-[0-9a-f]{4}-[0-9a-f]{12}/g, "UUID");
      s = s.replace(/"turnId":"[0-9a-f-]+"/g, '"turnId":"UUID"');
      return Buffer.from(s, "utf8");
    }

    async function captureSseBody(app: any, campaignId: string): Promise<Buffer> {
      const res = await app.request("/api/chat/action", {
        method: "POST",
        headers: { "content-type": "application/json" },
        body: JSON.stringify({ campaignId, action: "look", method: "do" }),
      });
      const reader = res.body!.getReader();
      const chunks: Uint8Array[] = [];
      while (true) {
        const { value, done } = await reader.read();
        if (done) break;
        chunks.push(value!);
      }
      return Buffer.concat(chunks);
    }

    describe("Storyteller SSE stream safety — byte-identical with observability off vs on", () => {
      let tmpDir: string;

      beforeEach(async () => {
        tmpDir = mkdtempSync(join(tmpdir(), "wf-obs-stream-"));
        process.env.GSD_LOG_ROOT = tmpDir;
        process.env.GSD_CAMPAIGNS_ROOT = tmpDir;  // route-level fixture sandbox (Plan 58-01)
        vi.resetModules();
        await applyMocks();
        const { resetLoggerForTest } = await import("../../lib/logger-setup.js");
        resetLoggerForTest({ logRoot: tmpDir });
      });

      afterEach(() => {
        delete process.env.GSD_LOG_ROOT;
        delete process.env.GSD_CAMPAIGNS_ROOT;
        vi.restoreAllMocks();
        vi.resetModules();
      });

      it("produces byte-identical SSE transcript with observability enabled vs disabled (Buffer.compare === 0)", async () => {
        const campaignId = "stream-safety";
        seedCampaignWithAllSeams(tmpDir, campaignId);

        const { default: app } = await import("../../app.js");
        const { configureObservability } = await import("../../lib/logger-setup.js");

        // Run 1 — observability disabled (baseline)
        configureObservability({
          enabled: false,
          dumpFullPrompts: false,
          roles: { judge: false, storyteller: false, oracle: false, npcAgent: false, reflection: false, embedder: false },
        });
        const baseline = normalizeSse(await captureSseBody(app, campaignId));

        // Run 2 — observability enabled (treatment)
        configureObservability({
          enabled: true,
          dumpFullPrompts: false,
          roles: { judge: true, storyteller: true, oracle: true, npcAgent: true, reflection: true, embedder: true },
        });
        const treatment = normalizeSse(await captureSseBody(app, campaignId));

        // BYTE-IDENTICAL after normalization
        expect(Buffer.compare(baseline, treatment)).toBe(0);
      });
    });
    ```
