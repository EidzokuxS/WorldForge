      setLastIngestion(() => callable);
      setIngestionError(null);
      try {
        await callable();
        setLastIngestion(null);
        setOverrideText("");   // clear override on success (research §Override Text State model — NPC path clears on success)

    === Phase C — 4 mode handlers ===

    Replace the current three handler functions (`handleParse`, `handleImport`, `handleGenerate` which is actually research) with FOUR handlers, each threading overrideText and setting busy. The new "generate" mode calls `generateCharacter(campaignId, "key", ...)` — this is pure AI-generate-from-scratch, DIFFERENT from research.

    ```typescript
    const handleParse = useCallback(() => {
      setBusy("parsing");
      return runIngestion(async () => {
        try {
          const result = await parseCharacter(campaignId, description, "key", locationNames, factionNames, overrideText);
          if (result.role === "key") onChange([...npcs, assignUid(result.npc)]);
          toast.success("NPC parsed");
          setDescription("");
        } finally { setBusy("idle"); }
      });
    }, [campaignId, description, locationNames, factionNames, overrideText, npcs, onChange]);

    const handleGenerate = useCallback(() => {
      setBusy("generating");
      return runIngestion(async () => {
        try {
          const result = await generateCharacter(campaignId, "key", locationNames, factionNames, overrideText);
          if (result.role === "key") onChange([...npcs, assignUid(result.npc)]);
          toast.success("NPC generated");
        } finally { setBusy("idle"); }
      });
    }, [campaignId, locationNames, factionNames, overrideText, npcs, onChange]);

    const handleResearch = useCallback(() => {
      setBusy("researching");
      return runIngestion(async () => {
        try {
          const result = await researchCharacter(campaignId, archetype, "key", locationNames, factionNames, overrideText);
          if (result.role === "key") onChange([...npcs, assignUid(result.npc)]);
          toast.success("Archetype researched");
          setArchetype("");
        } finally { setBusy("idle"); }
      });
    }, [campaignId, archetype, locationNames, factionNames, overrideText, npcs, onChange]);

    const handleImport = useCallback(async (file: File) => {
      setBusy("importing");
      return runIngestion(async () => {
        try {
          const card = await parseV2CardFile(file);
          const result = await importV2Card(campaignId, card, {
            role: "key",
            importMode,
            locationNames,
            factionNames,
            overrideText,
          });
          if (result.role === "key") onChange([...npcs, assignUid(result.npc)]);
          toast.success("V2 card imported");
        } finally { setBusy("idle"); }
      });
    }, [campaignId, importMode, locationNames, factionNames, overrideText, npcs, onChange]);
    ```
