```typescript
// Local draft state — edits happen here, propagated to parent on debounce
const [localDraft, setLocalDraft] = useState<CharacterDraft>(draft);
const onChangeRef = useRef(onChange);
onChangeRef.current = onChange;

// Sync from parent when draft identity changes (new parse/generate result)
useEffect(() => {
  setLocalDraft(draft);
}, [draft]);

// Debounced propagation to parent
const debounceRef = useRef<ReturnType<typeof setTimeout> | null>(null);

function update(next: CharacterDraft) {
  setLocalDraft(next);
  if (debounceRef.current) clearTimeout(debounceRef.current);
  debounceRef.current = setTimeout(() => {
    onChangeRef.current(next);
  }, 300);
}

// Cleanup on unmount
useEffect(() => {
  return () => {
    if (debounceRef.current) clearTimeout(debounceRef.current);
  };
}, []);
```
